36 /*
37  * MirBufferStream::egl_native_window() returns EGLNativeWindowType.
38  *
39  * EGLNativeWindowType is an EGL platform-specific type that is typically a
40  * (possibly slightly obfuscated) pointer. This makes our client module ABI
41  * technically EGL-platform dependent, which is awkward because we support
42  * multiple EGL platforms.
43  *
44  * On the Mesa platform EGLNativeWindow is a
45  * pointer or a uintptr_t.
46  *
47  * In practise EGLNativeWindowType is always a typedef to a pointer-ish, but
48  * for paranoia's sake make sure the build will fail if we ever encounter a
49  * strange EGL platform where this isn't the case.
50  */
51 #include <type_traits>
52 static_assert(
53  sizeof(EGLNativeWindowType) == sizeof(void*) &&
54  std::is_pod<EGLNativeWindowType>::value,
55  "The MirBufferStream requires that EGLNativeWindowType be no-op convertible to void*");
56 
57 #undef EGLNativeWindowType
58 #define EGLNativeWindowType void*
